common_skill_template:
  code_review:
    id: "code-review"
    name: "代码审查"
    description: "对代码变更进行自动化审查，检测潜在问题并提供改进建议"
    input:
      - target: "审查目标（文件路径或Git diff）"
      - review_type: "审查类型（quick/full/security）"
      - focus: "关注点（quality/security/performance/all）"
    output:
      - report: "审查报告"
      - issues: "问题列表"
      - suggestions: "改进建议"
    workflow:
      - step: "代码解析"
      - step: "静态分析"
      - step: "安全扫描"
      - step: "性能检查"
      - step: "最佳实践检查"
      - step: "报告生成"

  requirement_review:
    id: "requirement-review"
    name: "需求评审"
    description: "评审需求文档，检查完整性、可行性和风险"
    input:
      - requirement_doc: "需求文档路径"
      - review_scope: "评审范围"
    output:
      - review_report: "评审报告"
      - issues: "问题列表"
      - suggestions: "改进建议"
    workflow:
      - step: "文档解析"
      - step: "完整性检查"
      - step: "可行性分析"
      - step: "风险识别"
      - step: "报告生成"

  adr:
    id: "adr"
    name: "架构决策记录"
    description: "编写架构决策记录文档，记录重要的技术决策及其背景"
    input:
      - title: "决策标题"
      - context: "决策背景"
      - decision: "决策内容"
      - consequences: "决策后果"
    output:
      - adr_document: "ADR文档（Markdown格式）"
    template: |
      # ADR-{编号}: {标题}
      
      ## 状态
      {状态：提议/已接受/已废弃/已替代}
      
      ## 背景
      {描述导致此决策的背景和问题}
      
      ## 决策
      {描述所做的决策及其理由}
      
      ## 后果
      {描述决策带来的正面和负面影响}
      
      ## 替代方案
      {描述考虑过的其他方案及其拒绝理由}
