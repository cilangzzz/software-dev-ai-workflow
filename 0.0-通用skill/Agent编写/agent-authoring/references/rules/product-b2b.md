deliverables:
  requirement_phase:
    - name: "BRD（业务需求文档）"
      template: |
        # 业务需求文档
        
        ## 1. 业务背景
        {当前业务现状、痛点、机会}
        
        ## 2. 业务目标
        {定性目标、定量目标}
        
        ## 3. 用户分析
        {目标用户画像、使用场景}
        
        ## 4. 业务流程
        {现状流程、优化后流程}
        
        ## 5. 功能需求概述
        {核心功能列表}
        
        ## 6. 成本效益分析
        {投入评估、收益预测}

    - name: "PRD（产品需求文档）"
      template: |
        # 产品需求文档
        
        ## 1. 文档信息
        | 项目名称 | {name} |
        | 版本 | {version} |
        | 作者 | {author} |
        | 更新日期 | {date} |
        
        ## 2. 需求背景
        {需求来源、问题描述}
        
        ## 3. 功能需求
        ### 3.1 功能概述
        {功能列表}
        
        ### 3.2 功能详情
        | 功能名称 | 优先级 | 描述 | 验收标准 |
        
        ## 4. 业务流程
        {流程图 + 描述}
        
        ## 5. 页面设计
        {页面列表 + 交互说明}
        
        ## 6. 数据需求
        {数据埋点、指标定义}
        
        ## 7. 技术需求
        {性能要求、安全要求}
        
        ## 8. 验收标准
        {测试用例概要}

    - name: "用户故事"
      template: |
        # 用户故事
        
        ## 格式
        作为 {用户角色}
        我想要 {功能/操作}
        以便于 {业务价值/目的}
        
        ## 验收条件
        - Given: {前置条件}
        - When: {触发动作}
        - Then: {预期结果}
        
        ## 示例
        作为 采购经理
        我想要 快速创建采购订单
        以便于 提高采购效率，缩短采购周期
        
        验收条件:
        - Given: 用户已登录采购系统
        - When: 用户点击"新建采购订单"
        - Then: 系统展示采购订单创建表单
        - And: 用户可快速选择供应商和商品

  design_phase:
    - name: "功能列表"
      template: |
        # 功能列表
        
        | 功能编号 | 功能名称 | 优先级 | 状态 | 负责人 |
        
        ## P0 核心功能
        {核心功能详情}
        
        ## P1 重要功能
        {重要功能详情}
        
        ## P2 优化功能
        {优化功能详情}

    - name: "页面清单"
      template: |
        # 页面清单
        
        | 页面编号 | 页面名称 | 入口 | 功能说明 |
        
        ## 页面详情
        {每个页面的详细说明}

    - name: "验收标准"
      template: |
        # 验收标准
        
        ## 功能验收
        | 功能 | 验收条件 | 通过标准 |
        
        ## 性能验收
        {性能指标和标准}
        
        ## 安全验收
        {安全检查项}
