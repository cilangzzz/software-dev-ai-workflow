# ============================================
# 产出物规范
# ============================================
deliverables:
  ui_design:
    - name: "设计稿"
      format: "Figma/Sketch文件"
      content:
        - "页面设计稿"
        - "组件设计"
        - "样式定义"
        - "切图资源"

    - name: "设计规范文档"
      template: |
        # 设计规范
        
        ## 1. 色彩规范
        | 名称 | HEX值 | 使用场景 |
        
        ## 2. 字体规范
        | 字体名称 | 字号 | 行高 | 使用场景 |
        
        ## 3. 间距规范
        {间距定义}
        
        ## 4. 圆角规范
        {圆角定义}
        
        ## 5. 图标规范
        {图标设计指南}
        
        ## 6. 组件规范
        {组件设计规范}

    - name: "设计交付文档"
      template: |
        # 设计交付
        
        ## 页面清单
        | 页面名称 | 设计稿链接 | 状态 |
        
        ## 设计说明
        {关键设计说明}
        
        ## 资源清单
        {切图、图标资源}

  ux_design:
    - name: "用户研究报告"
      template: |
        # 用户研究报告
        
        ## 1. 研究背景
        {研究目的、问题定义}
        
        ## 2. 研究方法
        {访谈/问卷/测试方法}
        
        ## 3. 用户画像
        {用户特征、需求、痛点}
        
        ## 4. 研究发现
        {关键发现、问题}
        
        ## 5. 建议
        {设计建议、优化方向}

    - name: "交互设计文档"
      template: |
        # 交互设计
        
        ## 1. 页面概述
        {页面目的、用户场景}
        
        ## 2. 交互流程
        {流程图 + 说明}
        
        ## 3. 界面元素
        | 元素 | 类型 | 交互行为 | 说明 |
        
        ## 4. 状态设计
        {默认/加载/成功/失败状态}
        
        ## 5. 异常处理
        {边界情况处理}
        
        ## 6. 动效说明
        {动画和过渡定义}

    - name: "原型文件"
      format: "Figma/ProtoPie"
      content:
        - "交互原型"
        - "高保真原型"
        - "可用性测试原型"
